  def DBize.setup(id_field=nil)
    if id_field
      @@id_field = id_field
    else
      if @@dbi
        possible_ids = @@dbi.meta_names & ['id', 'recordno', 'recno', 'record']
        if possible_ids.empty?
          raise 'indeterminate id field'
        else
          @@id_field = possible_ids[0]
        end
      else
        raise 'indeterminate id field'
      end
    end
  end
