    def delete_from_database(rindex=nil)
      if rindex
        DBize.dbi.connection.transaction {
          self[rindex].delete_from_database
        }
      else
        DBize.dbi.connection.transaction {
          self.each { |r|
            r.delete_from_database
          }
        }
      end
    end
